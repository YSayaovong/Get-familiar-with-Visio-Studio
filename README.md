b) Using newline command : endl

 

#include <iostream>

using namespace std;

int main()
{
 

 

cout << "My first C++ program." << endl;
cout << "The sum of 2 and 3 = " << 5 << endl;
cout << "7 + 8 = " << 7 + 8 << endl;
return 0;

}
Notice that the endl command at the end of statement forces a newline.


Output: 

My first C++  program.
The sum of 2 and 3=5
7+8=15
